Objective

To design and implement a touchless, automatic smart dustbin that uses an ultrasonic proximity sensor to detect a user's hand and trigger a servo motor to open the lid — enabling hygienic, contact-free waste disposal without any physical interaction with the bin.

  • The HC-SR04 fires ultrasonic pulses every 200ms and measures the echo return time.
  • Arduino calculates distance using: Distance = (Duration × 0.034) / 2
  • If distance ≤ 15 cm → detected → servo rotates to 90° (lid opens).
  • Lid stays open for 3 seconds, then auto-closes (servo returns to 0°).
  • If distance > 15 cm → no detection → lid remains closed.

Arduino Source Code

Arduino C++ / Servo.h 
#include <Servo.h>

Servo lidServo;

// Pin definitions
const int trigPin   = 9;    // TRIG → Digital Pin 9
const int echoPin   = 10;   // ECHO → Digital Pin 10
const int servoPin  = 6;    // Servo → Digital Pin 6
const int threshold = 15;   // Detection threshold (cm)

long  duration;
int   distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  lidServo.attach(servoPin);
  lidServo.write(0);   // Initialise lid CLOSED
  Serial.begin(9600);
}

void loop() {
  // Send ultrasonic pulse
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  // Read echo and calculate distance
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  if (distance <= threshold) {
    lidServo.write(90);   // Open lid
    delay(3000);          // Hold open for 3 seconds
  } else {
    lidServo.write(0);    // Close lid
  }

  delay(200);  // Sampling rate
}
